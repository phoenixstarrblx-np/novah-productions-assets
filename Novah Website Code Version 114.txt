    async function onConfigChange(config) {
      const bgColor = config.background_color || defaultConfig.background_color;
      const surfaceColor = config.surface_color || defaultConfig.surface_color;
      const textColor = config.text_color || defaultConfig.text_color;
      const primaryColor = config.primary_action_color || defaultConfig.primary_action_color;
      const secondaryColor = config.secondary_action_color || defaultConfig.secondary_action_color;
      const fontFamily = config.font_family || defaultConfig.font_family;
      const fontSize = config.font_size || defaultConfig.font_size;

      document.body.style.backgroundColor = bgColor;
      document.body.style.color = textColor;
      document.body.style.fontFamily = `${fontFamily}, 'Courier New', monospace`;
      document.body.style.fontSize = `${fontSize}px`;

      document.getElementById('header').style.background = `linear-gradient(135deg, ${bgColor}, ${secondaryColor})`;
      document.getElementById('nav').style.backgroundColor = surfaceColor;
      document.getElementById('main').style.backgroundColor = bgColor;

      document.getElementById('site-title').textContent = config.site_title || defaultConfig.site_title;
      document.getElementById('site-title').style.fontSize = `${fontSize * 2}px`;
      document.getElementById('site-title').style.color = textColor;
      document.getElementById('site-title').style.fontFamily = `${fontFamily}, 'Courier New', monospace`;
      
      document.getElementById('tagline').textContent = config.tagline || defaultConfig.tagline;
      document.getElementById('tagline').style.fontSize = `${fontSize * 1.125}px`;
      document.getElementById('tagline').style.color = textColor;
      document.getElementById('tagline').style.fontFamily = `${fontFamily}, 'Courier New', monospace`;

      const tabButtons = document.querySelectorAll('.tab-btn');
      tabButtons.forEach(btn => {
        btn.style.color = textColor;
        btn.style.fontFamily = `${fontFamily}, 'Courier New', monospace`;
        btn.style.fontSize = `${fontSize}px`;
        btn.style.fontWeight = 'bold';
        btn.style.textShadow = `0 0 5px ${primaryColor}`;
        if (btn.classList.contains('active')) {
          btn.style.backgroundColor = primaryColor;
          btn.style.color = '#000000';
          btn.style.boxShadow = `0 0 20px ${primaryColor}`;
        } else {
          btn.style.backgroundColor = 'transparent';
          btn.style.boxShadow = 'none';
        }
      });

      document.getElementById('tab-btn-bio').textContent = config.tab_bio || defaultConfig.tab_bio;
      document.getElementById('tab-btn-community').textContent = config.tab_community || defaultConfig.tab_community;
      document.getElementById('tab-btn-games').textContent = config.tab_games || defaultConfig.tab_games;
      document.getElementById('tab-btn-events').textContent = config.tab_events || defaultConfig.tab_events;
      document.getElementById('tab-btn-shop').textContent = config.tab_shop || defaultConfig.tab_shop;
      document.getElementById('tab-btn-careers').textContent = config.tab_careers || defaultConfig.tab_careers;
      document.getElementById('tab-btn-gallery').textContent = config.tab_gallery || defaultConfig.tab_gallery;
      document.getElementById('tab-btn-updates').textContent = config.tab_updates || defaultConfig.tab_updates;
      document.getElementById('tab-btn-vault').textContent = config.tab_vault || defaultConfig.tab_vault;

      const headings = document.querySelectorAll('h2, h3, h4');
      headings.forEach(h => {
        h.style.color = textColor;
        h.style.fontFamily = `${fontFamily}, 'Courier New', monospace`;
        h.style.fontWeight = 'bold';
        if (h.tagName === 'H2') {
          h.style.fontSize = `${fontSize * 1.875}px`;
        } else if (h.tagName === 'H3') {
          h.style.fontSize = `${fontSize * 1.25}px`;
        } else {
          h.style.fontSize = `${fontSize * 1.125}px`;
        }
      });

      const cards = document.querySelectorAll('.game-card, .event-card, .career-card, .vault-card, .update-card, .photo-item');
      cards.forEach(card => {
        card.style.backgroundColor = surfaceColor;
        card.style.color = textColor;
      });

      const communityCards = document.querySelectorAll('#community > div > div, #bio > div > div');
      communityCards.forEach(card => {
        card.style.backgroundColor = surfaceColor;
        card.style.color = textColor;
      });

      const buttons = document.querySelectorAll('button:not(.tab-btn)');
      buttons.forEach(btn => {
        btn.style.backgroundColor = primaryColor;
        btn.style.color = '#000000';
        btn.style.fontFamily = `${fontFamily}, 'Courier New', monospace`;
        btn.style.fontSize = `${fontSize}px`;
        btn.style.fontWeight = 'bold';
        btn.style.border = `2px solid ${primaryColor}`;
        btn.style.boxShadow = `0 0 10px ${primaryColor}`;
      });

      const skillTags = document.querySelectorAll('.career-card span');
      skillTags.forEach(tag => {
        tag.style.backgroundColor = secondaryColor;
        tag.style.color = textColor;
        tag.style.fontSize = `${fontSize * 0.875}px`;
      });
    }
